
				m_dragButton = mouseEvent->button();
				m_dragStartWorld = hitOnTogo;
				m_prevHitWorld = hitOnTogo;
				m_dragStartRedPlane = m_redPlane;
				m_dragStartGreenPlane = m_greenPlane;
				m_useFirstHitpointForDrag = true;

			// Capture stable line direction and original plane normal at drag start
			// to prevent sign oscillation from line recomputation during drag.

		// Non-dragging move events and other events: don't consume
		return false;
	}
